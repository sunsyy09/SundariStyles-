        function sortProducts(type) {

            let sorted = [...products]

            if (type === 1) sorted.sort((a, b) => a.price - b.price)

            else sorted.sort((a, b) => b.price - a.price)

        }

        function addToCart(id) {

            const product = products.find(p => p.id === id)

            if (!product || product.stock < 1) { alert("Sorry, out of stock!"); return }

            const existing = cart.find(item => item.id === id)

            if (existing) existing.quantity = (existing.quantity || 1) + 1

            else cart.push({...product, quantity: 1})

            saveData()

            updateCartCount()

            showToast(`${product.name} added to cart!`)

        }

        function renderCart() {

            const container = document.getElementById('cart-items')

            if (cart.length === 0) {

                container.innerHTML = `<div class="bg-white rounded-3xl p-16 text-center text-2xl font-light">Your cart is empty 🌸<br><span onclick="navigateTo('shop')" class="text-[#FF1493] cursor-pointer text-xl">Start shopping →</span></div>`

                document.getElementById('cart-subtotal').innerHTML = '₹0'

                document.getElementById('cart-shipping').innerHTML = `₹${shippingCharge}`

                document.getElementById('cart-total').innerHTML = `₹${shippingCharge}`

                return

            }

            let subtotal = 0

            let html = ''

            cart.forEach((item, index) => {

                const itemTotal = item.price * (item.quantity || 1)

                subtotal += itemTotal

                html += `

                <div class="flex gap-6 bg-white rounded-3xl p-6 items-center">

                    <img src="${item.image}" class="w-24 h-24 object-cover rounded-2xl">

                    <div class="flex-1">

                        <div class="flex justify-between">

                            <div>

                                <div class="font-semibold">${item.name}</div>

                                <div class="font-mono text-xs text-gray-400">${item.sku}</div>

                            </div>

                            <div class="text-right">

                                <div class="font-semibold text-2xl">₹${itemTotal}</div>

                            </div>

                        </div>

                        <div class="flex justify-between mt-6">

                            <div class="flex items-center border rounded-3xl">

                                <button onclick="changeCartQty(${index}, -1)" class="px-5 py-2 text-xl">-</button>

                                <span class="px-6 font-semibold">${item.quantity || 1}</span>

                                <button onclick="changeCartQty(${index}, 1)" class="px-5 py-2 text-xl">+</button>

                            </div>

                            <button onclick="removeFromCart(${index})" class="text-red-400 hover:text-red-600 text-xl"><i class="fa-solid fa-trash"></i></button>

                        </div>

                    </div>

                </div>`

            })

            container.innerHTML = html

            document.getElementById('cart-subtotal').innerHTML = `₹${subtotal}`

            document.getElementById('cart-shipping').innerHTML = `₹${shippingCharge}`

            document.getElementById('cart-total').innerHTML = `₹${subtotal + shippingCharge}`

        }

        function changeCartQty(index, change) {

            const item = cart[index]

            const newQty = (item.quantity || 1) + change

            if (newQty < 1) { removeFromCart(index); return }

            item.quantity = newQty

            saveData()

            renderCart()

        }

        function removeFromCart(index) {

            cart.splice(index, 1)

            saveData()

            updateCartCount()

            renderCart()

        }

        function updateCartCount() {

            let count = 0

            cart.forEach(item => count += (item.quantity || 1))

            document.getElementById('cart-count-badge').textContent = count

        }

        function toggleWishlist(id) {

            const index = wishlist.findIndex(w => w.id === id)

            if (index > -1) wishlist.splice(index, 1)

            else {

                const product = products.find(p => p.id === id)

                if (product) wishlist.push(product)

            }

            saveData()

            updateWishlistCount()

            if (!document.getElementById('wishlist-page').classList.contains('hidden')) renderWishlist()

            if (!document.getElementById('shop-page').classList.contains('hidden')) renderShop()

        }

        function updateWishlistCount() {

            document.getElementById('wishlist-count-badge').textContent = wishlist.length

        }

        function renderWishlist() {

            const container = document.getElementById('wishlist-grid')

        function handleCheckout(e) {

            e.preventDefault()

            if (cart.length === 0) return

            const isPrepaid = document.getElementById('prepaid-radio').checked

            const advance = isPrepaid ? 0 : 100

            const btn = document.getElementById('place-order-btn')

            const originalText = btn.innerHTML

            btn.innerHTML = `<span class="animate-pulse">PROCESSING PAYMENT...</span>`

            btn.disabled = true

            setTimeout(() => {

                const order = {

                    id: 'ORD-' + Date.now().toString().slice(-6),

                    date: new Date().toLocaleString('en-IN'),

                    customer: {

                        name: document.getElementById('customer-name').value,

                        phone: document.getElementById('customer-phone').value,

                        address: document.getElementById('customer-address').value,

                        pincode: document.getElementById('customer-pincode').value,

                        city: document.getElementById('customer-city').value,

                        email: document.getElementById('customer-email').value

                    },

                    items: [...cart],

                    subtotal: cart.reduce((sum, item) => sum + item.price * (item.quantity || 1), 0),

                    shipping: shippingCharge,

                    paymentType: isPrepaid ? 'Prepaid (Razorpay)' : 'COD (₹100 advance paid)',

                    advancePaid: advance,

                    status: 'Confirmed'

                }

                orders.unshift(order)

                cart.forEach(cartItem => {

                    const prod = products.find(p => p.id === cartItem.id)

                    if (prod) prod.stock = Math.max(0, prod.stock - (cartItem.quantity || 1))

                })

                saveData()

                cart = []

                saveData()

                updateCartCount()

                btn.innerHTML = originalText

                btn.disabled = false

                const msg = isPrepaid ? `🎉 Order #\( {order.id} confirmed! Full payment received.` : `🎉 Order # \){order.id} confirmed! ₹100 advance paid. Balance on delivery.`

                document.getElementById('success-message').innerHTML = msg

                document.getElementById('success-modal').classList.remove('hidden')

                document.getElementById('success-modal').classList.add('flex')

                document.getElementById('checkout-form').reset()

            }, 1650)

        }

            } else {

        }
